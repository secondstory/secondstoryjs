/**
 * @class Ext.form.UrlField
 * @extends Ext.form.TextField
 * Wraps an HTML5 url field. See {@link Ext.form.FormPanel FormPanel} for example usage.
 * @xtype urlfield
 */

    inputType: 'url',    
    autoCapitalize : false
});
